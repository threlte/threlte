### Example

```svelte
<script>
  import { Line2 } from '@threlte/core'
  import { LineMaterial } from 'three/examples/jsm/lines/LineMaterial'
</script>

<Line2
  points={[
    [0, 1, 0],
    [0, 2, 1],
    [-1, 1, 4]
  ]}
  material={new LineMaterial({
    worldUnits: true,
    lineWidth: 0.2
  })}
/>
```
