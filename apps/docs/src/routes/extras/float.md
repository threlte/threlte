### Examples

#### Basic Example

```svelte
<script lang="ts">
  import { Mesh } from '@threlte/core'
  import { BoxBufferGeometry, MeshBasicMaterial } from 'three'
  import { Float } from '@threlte/extras'
</script>

<Float speed={2}>
  <Mesh geometry={new BoxBufferGeometry(1, 1, 1)} material={new MeshBasicMaterial()} />
</Float>
```
